package com.example.middleex;

import android.content.Context;
import android.content.Intent;
import android.os.Bundle;
import android.view.View;
import android.widget.Button;
import android.widget.ImageView;
import android.widget.ListView;

import androidx.appcompat.app.AppCompatActivity;
import androidx.core.content.ContextCompat;

public class ResultActivity extends AppCompatActivity {
    Button exit;
    ListView lvMain;
    CustomAdapter adapter;

    @Override
    protected void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        setContentView(R.layout.result_main);

        Intent get = getIntent();
        String picName[] = get.getStringArrayExtra("name");
        int count[] = get.getIntArrayExtra("count");
        int pic[] = {R.drawable.pic1, R.drawable.pic1, R.drawable.pic1, R.drawable.pic1, R.drawable.pic1, R.drawable.pic1, R.drawable.pic1, R.drawable.pic1, R.drawable.pic1};

        lvMain = (ListView)findViewById(R.id.lvMain);
        adapter = new CustomAdapter();
        lvMain.setAdapter(adapter);

        for(int i = 0; i < count.length; i++) {
            adapter.addItem(ContextCompat.getDrawable(this, pic[i]), picName[i], (float)count[i]);

        }

        exit = (Button)findViewById(R.id.btnExit);
        exit.setOnClickListener(new View.OnClickListener() {
            @Override
            public void onClick(View v) {
                finish();
            }
        });



    }
}
